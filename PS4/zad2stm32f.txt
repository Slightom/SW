/**
  * @brief  This function handles SysTick Handler.
  * @param  None
  * @retval None
  */


// konwersja na kod wyswietlacza 7-segmentowego
uint16_t int27seg(uint8_t cyfra)
{
	uint16_t result;
	switch (cyfra){
		case 0: result=0xC0; break; // 11000000
		case 1: result=0xF9; break; // 11111001
		case 2: result=0xA4; break; // 10100100
		case 3: result=0xB0; break; // 10110000
		case 4: result=0x99; break; // 10011001
		case 5: result=0x92; break; // 10010010
		case 6: result=0x82; break; // 10000010
		case 7: result=0xF8; break; // 11111000
		case 8: result=0x80; break; // 10000000
		case 9: result=0x90; break; // 10010000
		case 10: result=0x88; break;
		case 11: result=0x83; break;
		case 12: result=0xC6; break;
		case 13: result=0xA1; break;
		case 14: result=0x86; break;
		case 15: result=0x8E; break;
		default: result=0xFF; break; // 11111111 - nic do wyswietlenia
		}
	return result;
		
}


void SysTick_Handler(void)
{	
	if(counter == 0)
	{
	GPIO_Write(GPIOB, int27seg(1)); // zapal segmenty
	GPIO_ResetBits(GPIOB, GPIO_Pin_11); // wlacz wyswietlacz DS3
	GPIO_SetBits(GPIOB, GPIO_Pin_8 | GPIO_Pin_9 | GPIO_Pin_10); // wylacz pozostale
	}
	else	
	if(counter ==1)
	{
	GPIO_Write(GPIOB, int27seg(2)); // zapal segmenty
	GPIO_ResetBits(GPIOB, GPIO_Pin_10); // wlacz wyswietlacz DS2
	GPIO_SetBits(GPIOB, GPIO_Pin_8 | GPIO_Pin_9 | GPIO_Pin_11); // wylacz pozostale
	}
	else
	if(counter == 2)
	{
	GPIO_Write(GPIOB, int27seg(3)); // zapal segmenty
	GPIO_ResetBits(GPIOB, GPIO_Pin_9); // wlacz wyswietlacz DS1
	GPIO_SetBits(GPIOB, GPIO_Pin_8 | GPIO_Pin_10 | GPIO_Pin_11); // wylacz pozostale
	}
	else
	if(counter == 3)
	{
	GPIO_Write(GPIOB, int27seg(4)); // zapal segmenty
	GPIO_ResetBits(GPIOB, GPIO_Pin_8); // wlacz wyswietlacz DS0
	GPIO_SetBits(GPIOB, GPIO_Pin_9 | GPIO_Pin_10 | GPIO_Pin_11); // wylacz pozostale
	}
	counter++;
	if(counter == 4)
		counter = 0;

}
